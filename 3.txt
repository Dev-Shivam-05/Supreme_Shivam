    // === Audio Engine (Simple Drone) ===
    let audioCtx, oscillators = [];
    function startAudio(){
       if(audioCtx) return;
       try {
          audioCtx = new (window.AudioContext || window.webkitAudioContext)();
          const createOsc = (freq, type, gainVal) => {
             const osc = audioCtx.createOscillator();
             const gain = audioCtx.createGain();
             osc.type = type;
             osc.frequency.value = freq;
             gain.gain.value = gainVal;
             osc.connect(gain);
             gain.connect(audioCtx.destination);
             osc.start();
             oscillators.push({osc, gain});
          };
          
          // Deep drone
          createOsc(50, 'sine', 0.1);
          createOsc(52, 'sine', 0.1); // Binaural beat
          createOsc(100, 'triangle', 0.02); // Texture
       } catch(e) { console.error(e); }
    }
    
    function stopAudio(){
       if(!audioCtx) return;
       oscillators.forEach(o => {
          try{ o.osc.stop(); o.osc.disconnect(); }catch(e){}
       });
       oscillators = [];
       audioCtx.close();
       audioCtx = null;
    }

    // === Canvas Radar (Visual only) ===
    function initRadar(){
       const c = $('#radarCanvas');
       const ctx = c.getContext('2d');
       let angle = 0;
       
       const draw = () => {
          const w = c.width = c.offsetWidth;
          const h = c.height = c.offsetHeight;
          const cx = w/2, cy = h/2;
          
          ctx.clearRect(0,0,w,h);
          
          // Grid
          ctx.strokeStyle = 'rgba(255,255,255,0.1)';
          ctx.beginPath();
          ctx.arc(cx, cy, h*0.3, 0, Math.PI*2);
          ctx.stroke();
          ctx.beginPath();
          ctx.arc(cx, cy, h*0.15, 0, Math.PI*2);
          ctx.stroke();
          
          // Line
          ctx.save();
          ctx.translate(cx, cy);
          ctx.rotate(angle);
          ctx.strokeStyle = '#4ade80';
          ctx.lineWidth = 2;
          ctx.beginPath();
          ctx.moveTo(0,0);
          ctx.lineTo(h*0.4, 0);
          ctx.stroke();
          
          // Sweep gradient
          const grad = ctx.createConicGradient(0, 0, 0);
          grad.addColorStop(0, 'rgba(74, 222, 128, 0)');
          grad.addColorStop(0.8, 'rgba(74, 222, 128, 0)');
          grad.addColorStop(1, 'rgba(74, 222, 128, 0.2)');
          ctx.fillStyle = grad;
          ctx.beginPath();
          ctx.moveTo(0,0);
          ctx.arc(0, 0, h*0.4, 0, Math.PI*2);
          ctx.fill();
          
          ctx.restore();
          
          angle += 0.05;
          requestAnimationFrame(draw);
       };
       requestAnimationFrame(draw);
    }
