    function initParticles(){
      const c = $('#bgCanvas');
      const ctx = c.getContext('2d', { alpha: true });
      const DPR = Math.min(2, window.devicePixelRatio || 1);

      const state = {
        w: 0, h: 0,
        mx: innerWidth*0.5,
        my: innerHeight*0.35,
        t: 0,
        particles: [],
      };

      function resize(){
        state.w = Math.floor(innerWidth * DPR);
        state.h = Math.floor(innerHeight * DPR);
        c.width = state.w;
        c.height = state.h;
        c.style.width = innerWidth+'px';
        c.style.height = innerHeight+'px';
      }
      window.addEventListener('resize', resize, { passive:true });
      resize();

      window.addEventListener('mousemove', (e)=>{
        state.mx = e.clientX*DPR;
        state.my = e.clientY*DPR;
      }, { passive:true });

      const count = matchMedia('(max-width: 640px)').matches ? 60 : 110;
      for(let i=0;i<count;i++){
        state.particles.push(spawn(i));
      }

      function spawn(i){
        const sp = 0.25 + Math.random()*0.9;
        return {
          x: Math.random()*state.w,
          y: Math.random()*state.h,
          vx: (Math.random()*2-1)*sp,
          vy: (Math.random()*2-1)*sp,
          r: 0.8 + Math.random()*2.2,
          hue: 260 + Math.random()*80,
          a: 0.45 + Math.random()*0.35,
          id: i
        };
      }

      function draw(){
        state.t += 0.008;
        ctx.clearRect(0,0,state.w,state.h);

        // faint nebula
        const g = ctx.createRadialGradient(state.w*0.2, state.h*0.1, 0, state.w*0.2, state.h*0.1, Math.max(state.w,state.h)*0.85);
        g.addColorStop(0, 'rgba(185,103,255,0.12)');
        g.addColorStop(0.35, 'rgba(5,217,232,0.06)');
        g.addColorStop(1, 'rgba(0,0,0,0)');
        ctx.fillStyle = g;
        ctx.fillRect(0,0,state.w,state.h);

        // connections
        for(const p of state.particles){
          // drift + mouse gravity
          const dxm = (state.mx - p.x);
          const dym = (state.my - p.y);
          const d2 = dxm*dxm + dym*dym;
          if(d2 < (240*DPR)*(240*DPR)){
            const f = (1 - d2/((240*DPR)*(240*DPR))) * 0.015;
            p.vx += dxm * f * 0.001;
            p.vy += dym * f * 0.001;
          }

          p.x += p.vx;
          p.y += p.vy;
          p.vx *= 0.995;
          p.vy *= 0.995;

          if(p.x < -50) p.x = state.w+50;
          if(p.x > state.w+50) p.x = -50;
          if(p.y < -50) p.y = state.h+50;
          if(p.y > state.h+50) p.y = -50;
        }

        // draw links
        for(let i=0;i<state.particles.length;i++){
          const a = state.particles[i];
          for(let j=i+1;j<state.particles.length;j++){
            const b = state.particles[j];
            const dx = a.x-b.x, dy = a.y-b.y;
            const dist = Math.sqrt(dx*dx+dy*dy);
            const max = 120*DPR;
            if(dist < max){
              const alpha = (1 - dist/max) * 0.22;
              ctx.strokeStyle = `rgba(5,217,232,${alpha})`;
              ctx.lineWidth = 1;
              ctx.beginPath();
              ctx.moveTo(a.x,a.y);
              ctx.lineTo(b.x,b.y);
              ctx.stroke();
            }
          }
        }

        // draw particles
        for(const p of state.particles){
          const pulse = 0.4 + 0.6*Math.sin(state.t + p.id*0.05);
          ctx.fillStyle = `hsla(${p.hue}, 95%, 70%, ${p.a*0.65})`;
          ctx.beginPath();
          ctx.arc(p.x, p.y, p.r + pulse*0.3, 0, Math.PI*2);
          ctx.fill();
        }

        requestAnimationFrame(draw);
      }
      draw();
    }

    function initRadar(){
      const c = $('#radarCanvas');

          ctx.stroke();

          const a = (k/10)*Math.PI*2;
          ctx.beginPath();
          ctx.moveTo(cx,cy);
          ctx.lineTo(cx+Math.cos(a)*R, cy+Math.sin(a)*R);
          ctx.stroke();

        const grad = ctx.createRadialGradient(cx,cy,0,cx,cy,R);

    // === Audio (WebAudio, no external file) ===
    let audio = null;
    function startAudio(){
      if(audio) return;
      try{
        const ctx = new (window.AudioContext || window.webkitAudioContext)();
        const master = ctx.createGain();
        master.gain.value = 0.22;
        master.connect(ctx.destination);

        // noise
        const bufferSize = 2 * ctx.sampleRate;
        const noiseBuffer = ctx.createBuffer(1, bufferSize, ctx.sampleRate);
        const output = noiseBuffer.getChannelData(0);
        for(let i=0;i<bufferSize;i++) output[i] = (Math.random()*2-1);
        const noise = ctx.createBufferSource();
        noise.buffer = noiseBuffer;
        noise.loop = true;

        const hp = ctx.createBiquadFilter();
        hp.type = 'highpass';
        hp.frequency.value = 700;

        const lp = ctx.createBiquadFilter();
        lp.type = 'lowpass';
        lp.frequency.value = 2600;

        const noiseGain = ctx.createGain();
        noiseGain.gain.value = 0.035;

        noise.connect(hp); hp.connect(lp); lp.connect(noiseGain); noiseGain.connect(master);

        // sub drone
        const osc = ctx.createOscillator();
        osc.type = 'sine';
        osc.frequency.value = 52;

        const oscGain = ctx.createGain();
        oscGain.gain.value = 0.05;

        const wob = ctx.createOscillator();
        wob.type = 'sine';
        wob.frequency.value = 0.08;

        const wobGain = ctx.createGain();
        wobGain.gain.value = 14;

        wob.connect(wobGain);
        wobGain.connect(osc.frequency);

        osc.connect(oscGain); oscGain.connect(master);

        // soft clicky beeps (very subtle)
        const tickOsc = ctx.createOscillator();
        tickOsc.type = 'triangle';
        tickOsc.frequency.value = 880;
        const tickGain = ctx.createGain();
        tickGain.gain.value = 0;
        tickOsc.connect(tickGain); tickGain.connect(master);

        function randomTick(){
          if(!audio) return;
          const t = ctx.currentTime;
          const v = (Math.random() < 0.55) ? 0 : 0.04;
          tickGain.gain.cancelScheduledValues(t);
          tickGain.gain.setValueAtTime(0, t);
          tickGain.gain.linearRampToValueAtTime(v, t+0.01);
          tickGain.gain.exponentialRampToValueAtTime(0.0001, t+0.06);
          tickOsc.frequency.setValueAtTime(440 + Math.random()*700, t);
          setTimeout(randomTick, 700 + Math.random()*1400);
        }

        noise.start();
        osc.start();
        wob.start();
        tickOsc.start();
        randomTick();

        audio = { ctx, master, noise, osc, wob, tickOsc };
      }catch(e){
        console.warn('Audio init failed', e);
      }
    }

    function stopAudio(){
      try{
        if(!audio) return;
        const { ctx, noise, osc, wob, tickOsc } = audio;
        noise.stop(); osc.stop(); wob.stop(); tickOsc.stop();
        ctx.close();
      }catch{}
      audio = null;
    }
